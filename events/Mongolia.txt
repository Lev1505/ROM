country_event = {
	id = mongolia.12
	title = mongolia.12.t
	desc = mongolia.12.d
	is_triggered_only = yes
	
	option = { #Choibalsan
		trigger = { has_country_leader = { ruling_only = yes name = "Khorlogiin Choibalsan" } }
		name = mongolia.12.b
		add_stability = -0.1
		add_political_power = -100
		if = {
			limit = {
				AND = {
					check_variable = { choibalsan_influence > 74 }
					check_variable = { choibalsan_influence < 100 }
				}
			}
			random_owned_controlled_state = {
				damage_building = { type = industrial_complex damage = 0.5 }
				damage_building = { type = arms_factory damage = 0.5 }
				damage_building = { type = infrastructure damage = 0.5 }
			}
		}
		if = {
			limit = {
				AND = {
					check_variable = { choibalsan_influence > 49 }
					check_variable = { choibalsan_influence < 75 }
				}
			}
			random_owned_controlled_state = {
				damage_building = { type = industrial_complex damage = 1 }
				damage_building = { type = arms_factory damage = 1 }
				damage_building = { type = infrastructure damage = 1 }
			}
		}
		if = {
			limit = {
				AND = {
					check_variable = { choibalsan_influence > 24 }
					check_variable = { choibalsan_influence < 50 }
				}
			}
			random_owned_controlled_state = {
				damage_building = { type = industrial_complex damage = 1.5 }
				damage_building = { type = arms_factory damage = 1.5 }
				damage_building = { type = infrastructure damage = 1.5 }
			}
		}
		if = {
			limit = {
				check_variable = { choibalsan_influence < 25 }
			}
			random_owned_controlled_state = {
				damage_building = { type = industrial_complex damage = 2 }

				damage_building = { type = infrastructure damage = 2 }
			}
		}
	}
}
